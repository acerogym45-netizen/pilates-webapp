### 4️⃣ 중복 인물 필터링 기능

**목적**: 동일 이름 또는 전화번호로 중복 신청 확인

**구현 방법**:
```html
<!-- admin-main.html에 추가 -->
<button class="btn-check-duplicates" onclick="checkDuplicates()">
    <i class="fas fa-user-check"></i> 중복 확인
</button>
```

```javascript
// js/admin.js에 추가
function checkDuplicates() {
    const nameMap = {};
    const phoneMap = {};
    const duplicates = [];
    
    allContracts.forEach(contract => {
        // 이름 중복 체크
        if (nameMap[contract.name]) {
            nameMap[contract.name].push(contract);
        } else {
            nameMap[contract.name] = [contract];
        }
        
        // 전화번호 중복 체크
        if (phoneMap[contract.phone]) {
            phoneMap[contract.phone].push(contract);
        } else {
            phoneMap[contract.phone] = [contract];
        }
    });
    
    // 중복 발견
    Object.keys(nameMap).forEach(name => {
        if (nameMap[name].length > 1) {
            duplicates.push({
                type: '이름',
                value: name,
                contracts: nameMap[name]
            });
        }
    });
    
    Object.keys(phoneMap).forEach(phone => {
        if (phoneMap[phone].length > 1) {
            duplicates.push({
                type: '전화번호',
                value: phone,
                contracts: phoneMap[phone]
            });
        }
    });
    
    // 결과 표시
    if (duplicates.length === 0) {
        alert('✅ 중복된 신청이 없습니다.');
    } else {
        showDuplicatesModal(duplicates);
    }
}

function showDuplicatesModal(duplicates) {
    let html = '<h3>⚠️ 중복 신청 발견</h3>';
    
    duplicates.forEach(dup => {
        html += `<div class="duplicate-group">`;
        html += `<p><strong>${dup.type} 중복:</strong> ${dup.value}</p>`;
        html += `<ul>`;
        dup.contracts.forEach(c => {
            html += `<li>${c.name} (${c.dong} ${c.ho}) - ${c.lesson_type} - ${c.status}</li>`;
        });
        html += `</ul></div>`;
    });
    
    document.getElementById('duplicatesModalContent').innerHTML = html;
    document.getElementById('duplicatesModal').classList.add('active');
}
```
